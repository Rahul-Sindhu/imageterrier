41  /** 
42   * A split implementation that knows which it is in the scheme of things
43   * Based on the original Terrier PositionAwareSplit, but updated for Hadoop 0.20+
44   * 
45   * @author Jonathon Hare
46   * @author Richard McCreadie
47   * @param <T> the {@link FileSplit} type
48   */
49  public class PositionAwareSplitWrapper<T extends FileSplit & Writable> extends FileSplit implements Writable {
50  	/** the wrapped split */
51  	protected T split;
52  	
53  	/** the index of this split */
54  	protected int splitnum;
55  	
56  	/** Make a new split, for use in Writable serialization */
57  	public PositionAwareSplitWrapper() {
58  		super(null,0,0,null);
59  		splitnum=-1;
60  	}
61  	
62  	/** Make a new split with the specified attributs 
63  	 * @param _split 
64  	 * @param _splitnum 
65  	 */
66  	public PositionAwareSplitWrapper(T _split, int _splitnum) {
67  		super(null,0,0,null);
68  		split = _split;
69  		splitnum = _splitnum;
70  	}
71  
72  	/**
73  	 * Get the index of this split
74  	 * @return the splitnum
75  	 */
76  	public int getSplitIndex() {
77  		return splitnum;
78  	}
79  
80  	/**
81  	 * Set the index of this split
82  	 * @param splitnum the splitnum to set
83  	 */
84  	public void setSplitIndex(int splitnum) {
85  		this.splitnum = splitnum;
86  	}
87  
88  	/**
89  	 * Get the wrapped split
90  	 * @return the split
91  	 */
92  	public T getSplit() {
93  		return split;
94  	}
95  
96  	/**
97  	 * Set the wrapped split
98  	 * @param split the split to set
99  	 */
100 	public void setSplit(T split) {
101 		this.split = split;
102 	}
103 
104 	@Override
105 	public long getLength() {
106 		return split.getLength();
107 	}
108 
109 	@Override
110 	public String[] getLocations() throws IOException {
111 		return split.getLocations();
112 	}
113 
114 	@Override
115 	public Path getPath() {
116 		return split.getPath();
117 	}
118 
119 	@Override
120 	public long getStart() {
121 		return split.getStart();
122 	}
123 
124 	@Override
125 	public String toString() {
126 		return split.toString();
127 	}
128 
129 	@Override
130 	@SuppressWarnings("unchecked")
131 	public void readFields(DataInput in) throws IOException {
132 		try {
133 			final String className = in.readUTF();
134 			Class<?> c = Class.forName(className, false, this.getClass().getClassLoader());
135 			Constructor<?> constr = c.getDeclaredConstructor();
136 			constr.setAccessible(true);
137 			split = (T)constr.newInstance();
138 			split.readFields(in);
139 			splitnum = in.readInt();
140 		} catch (Exception e) {
141 			throw new WrappedIOException("Error during the reading of fields of a new PositionAwareSplitWrapper", e);
142 		} 
143 	}
144 
145 	/** {@inheritDoc} */
146 	@Override
147 	public void write(DataOutput out) throws IOException {
148 		out.writeUTF(split.getClass().getName());
149 		split.write(out);
150 		out.writeInt(splitnum);
151 	}
152 }
