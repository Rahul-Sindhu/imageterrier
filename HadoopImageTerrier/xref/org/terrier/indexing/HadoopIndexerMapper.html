112 	@Override
113 	protected void setup(Context context) throws IOException, InterruptedException {
114 		TerrierHDFSAdaptor.initialiseHDFSAdaptor(context.getConfiguration());
115 
116 		proxyIndexer = createIndexer(context);
117 
118 		currentContext = context;
119 		splitnum = getSplitNum(context);
120 
121 		proxyIndexer.setFlushDelegate(this);
122 
123 		final Path indexDestination = FileOutputFormat.getWorkOutputPath(context);
124 		indexDestination.getFileSystem(context.getConfiguration()).mkdirs(indexDestination);
125 
126 		mapTaskID = getTaskID(context);
127 		proxyIndexer.currentIndex = Index.createNewIndex(indexDestination.toString(), mapTaskID);
128 		proxyIndexer.maxMemory = Long.parseLong(ApplicationSetup.getProperty("indexing.singlepass.max.postings.memory",
129 				"0"));
130 
131 		// during reduce, we dont want to load indices into memory, as we only
132 		// use them as streams
133 		proxyIndexer.currentIndex.setIndexProperty("index.preloadIndices.disabled", "true");
134 		runData = new DataOutputStream(Files.writeFileStream(new Path(indexDestination, mapTaskID + ".runs").toString()));
135 		runData.writeUTF(mapTaskID);
136 
137 		proxyIndexer.createMemoryPostings();
138 		proxyIndexer.docIndexBuilder = new DocumentIndexBuilder(proxyIndexer.currentIndex, "document");
139 		proxyIndexer.metaBuilder = createMetaIndexBuilder();
140 		proxyIndexer.emptyDocIndexEntry = (FieldScore.FIELDS_COUNT > 0) ? new FieldDocumentIndexEntry(
141 				FieldScore.FIELDS_COUNT) : new SimpleDocumentIndexEntry();
142 
143 		final String filterFile = context.getConfiguration().get(HadoopIndexerOptions.DOCUMENT_ID_FILTER_FILE);
144 		synchronized (HadoopIndexerMapper.class) {
145 			if (filterFile != null) {
146 				if (filter == null) {
147 					logger.warn("Loading filter...");
148 					final Timer timer = Timer.timer();
149 					final FileSystem fs = FileSystem.get(context.getConfiguration());
150 					final Path p = new Path(filterFile);
151 					filter = new DocListFilter(fs, p);
152 					logger.warn("Took:" + timer.duration() / 1000f + "s");
153 				}
154 			}
155 		}
156 	}

178 	/**
179 	 * Map processes a single document. Stores the terms in the document along
180 	 * with the posting list until memory is full or all documents in this map
181 	 * have been processed then writes then to the output collector.
182 	 * 
183 	 * @param key
184 	 *            Wrapper for Document Identifier
185 	 * @param value
186 	 *            The document itself (a serialized list of quantised features)
187 	 * @param context
188 	 *            The mapper context
189 	 * @throws IOException
190 	 */
191 	@Override
192 	protected void map(Text key, VALUEIN value, Context context) throws IOException, InterruptedException {
193 		final String docno = key.toString();
194 		if (filter != null) {
195 			if (!filter.test(docno)) {
196 				context.getCounter(Counters.INDEXED_DOCUMENTS_SKIPPED).increment(1);
197 				return;
198 			}
199 		}
200 		context.setStatus("Currently indexing " + docno);
201 
202 		final Document doc = recordToDocument(key, value);
203 		if (doc == null)
204 			return;
205 
206 		indexDocument(doc, context);
207 		context.getCounter(Counters.INDEXED_DOCUMENTS).increment(1);
208 	}
209 
210 	protected void indexDocument(final Document doc, Context context) throws IOException {
211 		/* setup for parsing */
212 		proxyIndexer.createDocumentPostings();
213 
214 		String term;// term we're currently processing
215 		proxyIndexer.numOfTokensInDocument = 0;
216 
217 		// get each term in the document
218 		while (!doc.endOfDocument()) {
219 			context.progress();
220 
221 			if ((term = doc.getNextTerm()) != null && !term.equals("")) {
222 				proxyIndexer.termFields = doc.getFields();
223 
224 				// perform pre-op
225 				proxyIndexer.preProcess(doc, term); // JH MOD
226 
227 				/* pass term into TermPipeline (stop, stem etc) */
228 				proxyIndexer.pipeline_first.processTerm(term);
229 				/*
230 				 * the term pipeline will eventually add the term to this
231 				 * object.
232 				 */
233 			}
234 			if (proxyIndexer.MAX_TOKENS_IN_DOCUMENT > 0 &&
235 					proxyIndexer.numOfTokensInDocument > proxyIndexer.MAX_TOKENS_IN_DOCUMENT)
236 				break;
237 		}
238 
239 		// if we didn't index all tokens from document,
240 		// we need tocurrentId get to the end of the document.
241 		while (!doc.endOfDocument()) {
242 			doc.getNextTerm();
243 		}
244 		/*
245 		 * we now have all terms in the DocumentTree, so we save the document
246 		 * tree
247 		 */
248 		if (proxyIndexer.termsInDocument.getDocumentLength() == 0)
249 		{ /* this document is empty, add the minimum to the document index */
250 			proxyIndexer.indexEmpty(doc.getAllProperties());
251 			if (proxyIndexer.IndexEmptyDocuments)
252 			{
253 				proxyIndexer.currentId++;
254 				proxyIndexer.numberOfDocuments++;
255 				context.getCounter(Counters.INDEXED_EMPTY_DOCUMENTS).increment(1);
256 			}
257 		}
258 		else
259 		{ /* index this document */
260 			try {
261 				proxyIndexer.numberOfTokens += proxyIndexer.numOfTokensInDocument;
262 				proxyIndexer.indexDocument(doc.getAllProperties(), proxyIndexer.termsInDocument);
263 
264 				context.getCounter(Counters.INDEXED_TOKENS).increment(proxyIndexer.numOfTokensInDocument);
265 				context.getCounter(Counters.INDEXED_POINTERS).increment(
266 						proxyIndexer.termsInDocument.getNumberOfPointers());
267 			} catch (final IOException ioe) {
268 				throw ioe;
269 			} catch (final Exception e) {
270 				throw new WrappedIOException(e);
271 			}
272 		}
273 
274 		proxyIndexer.termsInDocument.clear();
275 	}

277 	/** causes the posting lists built up in memory to be flushed out */
278 	@Override
279 	public void forceFlush() throws IOException
280 	{
281 		ExtensibleSinglePassIndexer.logger.info("Map " + mapTaskID + ", flush requested, containing "
282 				+ proxyIndexer.numberOfDocsSinceFlush + " documents, flush " + flushNo);
283 
284 		if (proxyIndexer.mp == null)
285 			throw new IOException("Map flushed before any documents were indexed");
286 
287 		proxyIndexer.mp.finish(new NewHadoopRunWriter<VALUEIN>(currentContext, mapTaskID, splitnum, flushNo));
288 		runData.writeInt(proxyIndexer.currentId);
289 
290 		if (currentContext != null)
291 			currentContext.getCounter(Counters.INDEXER_FLUSHES).increment(1);
292 
293 		System.gc();
294 
295 		proxyIndexer.createMemoryPostings();
296 		proxyIndexer.memoryCheck.reset();
297 		proxyIndexer.numberOfDocsSinceFlush = 0;
298 		proxyIndexer.currentId = 0;
299 		flushNo++;
300 	}
301 
302 	@Override
303 	protected void cleanup(Context context) throws IOException, InterruptedException {
304 		proxyIndexer.forceFlush();
305 		proxyIndexer.docIndexBuilder.finishedCollections();
306 		proxyIndexer.currentIndex.setIndexProperty("index.inverted.fields.count", "" + FieldScore.FIELDS_COUNT);
307 		if (FieldScore.FIELDS_COUNT > 0)
308 		{
309 			proxyIndexer.currentIndex
310 					.addIndexStructure("document-factory", FieldDocumentIndexEntry.Factory.class.getName(),
311 							"java.lang.String", "${index.inverted.fields.count}");
312 		}
313 		else
314 		{
315 			proxyIndexer.currentIndex.addIndexStructure("document-factory",
316 					SimpleDocumentIndexEntry.Factory.class.getName(), "", "");
317 		}
318 		proxyIndexer.metaBuilder.close();
319 		proxyIndexer.currentIndex.flush();
320 		proxyIndexer.currentIndex.close();
321 		runData.writeInt(-1);
322 		runData.writeInt(proxyIndexer.numberOfDocuments);
323 		runData.writeInt(splitnum);
324 		runData.close();
325 		ExtensibleSinglePassIndexer.logger.info("Map " + mapTaskID + " finishing, indexed "
326 				+ proxyIndexer.numberOfDocuments + " in " + (flushNo - 1) + " flushes");
327 	}
