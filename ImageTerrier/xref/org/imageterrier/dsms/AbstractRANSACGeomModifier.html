57  /**
58   * Abstract base class for {@link DocumentScoreModifier}s that use RANSAC to fit
59   * a geometric model to matching pairs of visual terms.
60   * 
61   * Use of this class (or subclasses) requires that you are using a
62   * {@link PositionInvertedIndex} with a {@link PositionSpec} that has encoded
63   * spatial (x and y) coordinates for each term posting.
64   * 
65   * @author Jonathon Hare <jsh2@ecs.soton.ac.uk>
66   * 
67   */
68  public abstract class AbstractRANSACGeomModifier implements DocumentScoreModifier {
69  	public enum ScoringScheme {
70  		BINARY {
71  			@Override
72  			public double score(double originalScore, boolean didMatch, int nInliers, int nOutliers) {
73  				return (didMatch ? 1.0 : 0.0) * originalScore;
74  			}
75  		},
76  		MULTIPLY_NUM_MATCHES {
77  			@Override
78  			public double score(double originalScore, boolean didMatch, int nInliers, int nOutliers) {
79  				return (didMatch ? nInliers : 0.0) * originalScore;
80  			}
81  		},
82  		NUM_MATCHES {
83  			@Override
84  			public double score(double originalScore, boolean didMatch, int nInliers, int nOutliers) {
85  				return (didMatch ? nInliers : 0.0);
86  			}
87  		},
88  		MULTIPLY_PERCENTAGE_MATCHES {
89  			@Override
90  			public double score(double originalScore, boolean didMatch, int nInliers, int nOutliers) {
91  				return (didMatch ? (double) nInliers / (double) (nInliers + nOutliers) : 0.0) * originalScore;
92  			}
93  		},
94  		PERCENTAGE_MATCHES {
95  			@Override
96  			public double score(double originalScore, boolean didMatch, int nInliers, int nOutliers) {
97  				return (didMatch ? (double) nInliers / (double) (nInliers + nOutliers) : 0.0);
98  			}
99  		};
100 
101 		public abstract double score(double originalScore, boolean didMatch, int nInliers, int nOutliers);
102 	}

122 	@Override
123 	public boolean modifyScores(Index index, MatchingQueryTerms queryTerms, ResultSet resultSet) {
124 		final Lexicon<String> lexicon = index.getLexicon();
125 		final PositionInvertedIndex invertedIndex = (PositionInvertedIndex) index.getInvertedIndex();
126 		final PositionSpec spec = invertedIndex.getPositionSpec();
127 
128 		final double[] scores = resultSet.getScores();
129 		final QLFDocumentQuery<?> query = (QLFDocumentQuery<?>) queryTerms.getQuery();
130 		final QLFDocument<?> queryDoc = query.getDocument();
131 
132 		final int nd2r = ApplicationSetupUtils.getProperty(N_DOCS_TO_RERANK, resultSet.getResultSize());
133 		final int nRerankDocs = (nd2r <= 0 ? resultSet.getResultSize() : (nd2r > resultSet.getResultSize() ? resultSet
134 				.getResultSize() : nd2r));
135 
136 		queryDoc.reset();
137 
138 		// record matches per doc:
139 		final TIntObjectHashMap<List<Pair<Point2d>>> allMatchingPoints = new TIntObjectHashMap<List<Pair<Point2d>>>();
140 		String queryTerm;
141 		while (!queryDoc.endOfDocument()) {
142 			queryTerm = queryDoc.getNextTerm();
143 			final LexiconEntry le = lexicon.getLexiconEntry(queryTerm);
144 			if (le != null) {
145 				final int[] queryPos = spec.getPosition(queryDoc);
146 
147 				final TIntObjectHashMap<int[][]> matchDocs = invertedIndex.getPositions((BitIndexPointer) le);
148 
149 				for (int i = 0; i < nRerankDocs; i++) {
150 					final int docid = resultSet.getDocids()[i];
151 					final int[][] matchingPoints = matchDocs.get(docid);
152 
153 					if (matchingPoints != null) {
154 						List<Pair<Point2d>> ml = allMatchingPoints.get(docid);
155 						if (ml == null)
156 							allMatchingPoints.put(docid, ml = new ArrayList<Pair<Point2d>>(1));
157 
158 						addMatches(queryPos, matchingPoints, ml);
159 					}
160 				}
161 			}
162 		}
163 
164 		// now to do homography stuff
165 		final EstimatableModel<Point2d, Point2d> hm = makeModel();
166 
167 		final int nIter = ApplicationSetupUtils.getProperty(RANSAC_MAX_ITER, 100);
168 		final double perItemsSuccess = ApplicationSetupUtils.getProperty(RANSAC_PER_MATCHES_SUCCESS, 0.5);
169 
170 		final ScoringScheme scoringScheme = ScoringScheme.valueOf(ApplicationSetup.getProperty(SCORING_SCHEME,
171 				ScoringScheme.NUM_MATCHES.name()));
172 
173 		RANSAC.StoppingCondition stoppingCondition = null;
174 
175 		if (perItemsSuccess > 1) {
176 			stoppingCondition = new RANSAC.NumberInliersStoppingCondition((int) perItemsSuccess);
177 		} else if (perItemsSuccess > 0 && perItemsSuccess <= 1) {
178 			stoppingCondition = new RANSAC.PercentageInliersStoppingCondition(perItemsSuccess);
179 		} else if (perItemsSuccess < 0 && perItemsSuccess > -1) {
180 			stoppingCondition = new RANSAC.ProbabilisticMinInliersStoppingCondition(Math.abs(perItemsSuccess));
181 		} else {
182 			stoppingCondition = new RANSAC.BestFitStoppingCondition();
183 		}
184 
185 		@SuppressWarnings({ "rawtypes", "unchecked" })
186 		final RANSAC<Point2d, Point2d, ?> ransac = new RANSAC<Point2d, Point2d, EstimatableModel<Point2d, Point2d>>(hm,
187 				new AlgebraicResidual2d(), getTolerance(),
188 				nIter, stoppingCondition, false);
189 
190 		for (int i = 0; i < nRerankDocs; i++) {
191 			final int docid = resultSet.getDocids()[i];
192 			final List<Pair<Point2d>> data = allMatchingPoints.get(docid);
193 			filter(data);
194 
195 			final boolean didFit = ransac.fitData(data);
196 			final int nInliers = ransac.getInliers().size();
197 			final int nOutliers = ransac.getOutliers().size();
198 			scores[i] = scoringScheme.score(scores[i], didFit, nInliers, nOutliers);
199 		}
200 
201 		return true;
202 	}

211 	protected void filter(List<Pair<Point2d>> in) {
212 		final float thresh = ApplicationSetupUtils.getProperty(FILTERING_THRESHOLD, 0.5f);
213 
214 		if (thresh <= 0)
215 			return;
216 
217 		// remove matches where there is a big inbalance
218 		final TObjectIntHashMap<Point2d> forwardMatches = new TObjectIntHashMap<Point2d>();
219 		final TObjectIntHashMap<Point2d> reverseMatches = new TObjectIntHashMap<Point2d>();
220 
221 		for (final Pair<Point2d> m : in) {
222 			forwardMatches.adjustOrPutValue(m.firstObject(), 1, 1);
223 			reverseMatches.adjustOrPutValue(m.secondObject(), 1, 1);
224 		}
225 
226 		final List<Pair<Point2d>> remove = new ArrayList<Pair<Point2d>>();
227 		for (final Pair<Point2d> m : in) {
228 			final float score = (float) Math.min(forwardMatches.get(m.firstObject()),
229 					reverseMatches.get(m.secondObject()))
230 					/ (float) Math.max(forwardMatches.get(m.firstObject()), reverseMatches.get(m.secondObject()));
231 
232 			if (score < thresh)
233 				remove.add(m);
234 		}
235 
236 		in.removeAll(remove);
237 	}

242 	protected void addMatches(int[] qp, int[][] tps, List<Pair<Point2d>> matches) {
243 		final Point2d qpt = new Point2dImpl(qp[0], qp[1]);
244 
245 		for (final int[] tp : tps) {
246 			final Point2d tpt = new Point2dImpl(tp[0], tp[1]);
247 			matches.add(new Pair<Point2d>(qpt, tpt));
248 		}
249 	}
