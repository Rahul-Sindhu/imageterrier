150 	public <T extends QuantisedLocalFeature<?>> List<DocidScore> search(QLFDocument<T> query,
151 			BasicSearcherOptions options, Index index)
152 	{
153 		final QLFDocumentQuery<T> q = new QLFDocumentQuery<T>(query);
154 
155 		final Manager manager = new Manager(index);
156 		final SearchRequest request = manager.newSearchRequest("foo");
157 		request.setQuery(q);
158 		options.getMatchingModelType().configureRequest(request, q);
159 		ApplicationSetup.setProperty("matching.dsms",
160 				options.getScoreModifierTypeOptions().getScoreModifierClass(index.getInvertedIndex()));
161 
162 		ApplicationSetup.setProperty("ignore.low.idf.terms", "false");
163 
164 		if (options.getLimit() == 0)
165 			ApplicationSetup.setProperty("matching.retrieved_set_size", ""
166 					+ index.getCollectionStatistics().getNumberOfDocuments());
167 		else
168 			ApplicationSetup.setProperty("matching.retrieved_set_size", "" + options.getLimit());
169 
170 		manager.runPreProcessing(request);
171 		manager.runMatching(request);
172 		manager.runPostProcessing(request);
173 		manager.runPostFilters(request);
174 		final ResultSet rs = request.getResultSet();
175 		final int[] docids = rs.getDocids();
176 		final double[] scores = rs.getScores();
177 		final List<DocidScore> ret = new ArrayList<DocidScore>();
178 		for (int i = 0; i < docids.length; i++) {
179 			final DocidScore e = new DocidScore(docids[i], scores[i]);
180 			e.index = index;
181 			ret.add(e);
182 		}
183 		return ret;
184 	}

230 	public List<DocidScore> search(File imageFile, int[] coords, BasicSearcherOptions options, String featureType)
231 			throws Exception
232 	{
233 		// final Assigner<?> quantizer =
234 		// getQuantizer(options.getSoftQuantNeighbours());
235 		final Assigner<?> quantizer = hardAssigner;
236 
237 		final long t1 = System.currentTimeMillis();
238 
239 		// process the image
240 		final LocalFeatureMode mode = LocalFeatureMode.valueOf(featureType);
241 		final LocalFeatureList<?> features = FeatureTask.computeFeatures(imageFile, mode.getOptions());
242 
243 		LocalFeatureList<QuantisedLocalFeature<?>> qfeatures;
244 		if (options.getSoftQuantNeighbours() == 0)
245 			qfeatures = QuantiserTask.quantiseFeatures((HardAssigner<?, ?, ?>) quantizer, features);
246 		else
247 			qfeatures = QuantiserTask.quantiseFeaturesSoft((SoftAssigner<?, ?>) quantizer, features);
248 
249 		final QLFDocument<QuantisedLocalFeature<?>> d = new QLFDocument<QuantisedLocalFeature<?>>(qfeatures, "query",
250 				null);
251 
252 		if (coords != null) {
253 			final Rectangle r = new Rectangle(coords[0], coords[1], coords[2], coords[3]);
254 			d.filter(r);
255 		}
256 
257 		final long t2 = System.currentTimeMillis();
258 
259 		final List<DocidScore> ret = search(d, options);
260 
261 		final long t3 = System.currentTimeMillis();
262 
263 		if (options.timeQuery()) {
264 			System.out.println("[INFO] Feature extraction took:	" + ((t2 - t1) / 1000.0) + " secs");
265 			System.out.println("[INFO] Search took:				" + ((t3 - t2) / 1000.0) + " secs");
266 		}
267 
268 		return ret;
269 	}

271 	public void printResultSet(List<DocidScore> rs, int limit) throws IOException {
272 		if (limit <= 0)
273 			limit = rs.size();
274 
275 		for (final DocidScore didScore : rs) {
276 			final String namedDoc = this.pm.docidToString(didScore.index, didScore.first);
277 
278 			if (didScore.second <= 0)
279 				break; // filter 0 results
280 
281 			System.out.format("%s\t%f\n", namedDoc, didScore.second);
282 		}
283 	}

286 		if (limit <= 0)
287 			limit = rs.size();
288 		final List<File> files = new ArrayList<File>();
289 
290 		for (final DocidScore didScore : rs) {
291 			final File file = getFile(didScore.index, didScore.first);
292 
293 			if (didScore.second <= 0)
294 				break; // filter 0 results
295 
296 			files.add(file);
297 		}
298 
299 		displayImage(title, files.toArray(new File[files.size()]));
300 	}

326 	public static void main(String[] args) throws Exception {
327 		BasicSearcherOptions options = new BasicSearcherOptions();
328 		final CmdLineParser parser = new CmdLineParser(options);
329 
330 		try {
331 			parser.parseArgument(args);
332 		} catch (final CmdLineException e) {
333 			System.err.println(e.getMessage());
334 			System.err.println("Usage: java -jar ImageTerrier.jar [options...]");
335 			parser.printUsage(System.err);
336 			return;
337 		}
338 
339 		BasicSearcher<BasicSearcherOptions> searcher = new BasicSearcher<BasicSearcherOptions>(options);
340 		searcher.loadCompletely();
341 
342 		if (!options.isInteractive() && !options.isServer()) {
343 			if (options.getQueryImage() == null) {
344 				System.err.println("Error: No query image provided.");
345 				System.err.println("Usage: java -jar ImageTerrier.jar [options...]");
346 				parser.printUsage(System.err);
347 				return;
348 			}
349 
350 			if (options.getQueryImage().isDirectory()) {
351 				for (final File f : options.getQueryImage().listFiles()) {
352 					try {
353 						final List<DocidScore> rs = searcher.search(f, options.getRoiCoords(), options);
354 						System.out.println("Results from querying with: " + f.getName());
355 						searcher.printResultSet(rs, options.getLimit());
356 					} catch (final Exception e) {
357 						// ignore it
358 					}
359 				}
360 			} else {
361 				if (options.displayQuery())
362 					searcher.displayImage("Query:  " + options.getQueryImage(), options.getQueryImage());
363 
364 				final List<DocidScore> rs = searcher.search(options.getQueryImage(), options.getRoiCoords(), options);
365 				searcher.printResultSet(rs, options.getLimit());
366 
367 				if (options.displayResults())
368 					searcher.displayResults("Results:  " + options.getQueryImage(), rs, options.getLimit());
369 			}
370 		} else if (options.isServer()) {
371 			BasicSearcherXmlRpcServlet.options = options;
372 			BasicSearcherXmlRpcServlet.searcher = searcher;
373 			final XmlRpcServlet servlet = new XmlRpcServlet() {
374 				private static final long serialVersionUID = 1L;
375 
376 				@Override
377 				protected XmlRpcHandlerMapping newXmlRpcHandlerMapping() throws XmlRpcException {
378 					final PropertyHandlerMapping mapping = (PropertyHandlerMapping) super.newXmlRpcHandlerMapping();
379 					XmlRpcSystemImpl.addSystemHandler(mapping);
380 					return mapping;
381 				}
382 			};
383 			final ServletWebServer webServer = new ServletWebServer(servlet, options.serverPort());
384 			webServer.start();
385 		} else {
386 			// interactive mode -- useful for debugging
387 			// Note to self: jline is awesome!
388 			final ConsoleReader cr = new ConsoleReader();
389 			final ArgumentCompletor ac = new ArgumentCompletor(new FileNameCompletor());
390 			ac.setStrict(false);
391 			cr.addCompletor(ac);
392 			while (true) {
393 				String cmd = cr.readLine("query> ").trim();
394 
395 				if (cmd.length() == 0)
396 					continue;
397 
398 				if (cmd.equals("exit") || cmd.equals("quit"))
399 					break;
400 
401 				if (!cmd.contains("-i "))
402 					cmd += " -i " + options.getIndex();
403 				final BasicSearcherOptions interactiveOptions = new BasicSearcherOptions();
404 				final CmdLineParser interactiveParser = new CmdLineParser(interactiveOptions);
405 
406 				try {
407 					interactiveParser.parseArgument(cmd.split("\\s+"));
408 
409 					if (interactiveOptions.getQueryImage() == null) {
410 						System.out.println("Error: No query image specified.");
411 						continue;
412 					}
413 
414 					if (!interactiveOptions.getIndex().equals(options.getIndex())) {
415 						// reload index
416 						searcher = new BasicSearcher<BasicSearcherOptions>(interactiveOptions);
417 						options = interactiveOptions;
418 					}
419 
420 					// print a description of the query
421 					System.out.print("SELECT IMG_FILE, SCORE FROM " + interactiveOptions.getIndex() + " WHERE IMG");
422 					if (interactiveOptions.getScoreModifierType() != ScoreModifierType.NONE) {
423 						System.out.print(" CONTAINS ");
424 					} else {
425 						System.out.print(" IS_SIMILAR_TO ");
426 					}
427 					System.out.print(interactiveOptions.getQueryImage());
428 					if (interactiveOptions.getRoiCoordsString() != null)
429 						System.out.print("(" + interactiveOptions.getRoiCoordsString() + ")");
430 					if (interactiveOptions.getLimit() != 0) {
431 						System.out.print(" LIMIT " + interactiveOptions.getLimit());
432 					} else if (options.getLimit() != 0) {
433 						System.out.print(" LIMIT " + options.getLimit());
434 					}
435 					System.out.println();
436 
437 					int limit = options.getLimit();
438 					if (interactiveOptions.getLimit() != 0)
439 						limit = interactiveOptions.getLimit();
440 
441 					if (interactiveOptions.displayQuery())
442 						searcher.displayImage("Query:  " + interactiveOptions.getQueryImage(),
443 								interactiveOptions.getQueryImage());
444 
445 					final List<DocidScore> rs = searcher.search(interactiveOptions.getQueryImage(),
446 							interactiveOptions.getRoiCoords(), interactiveOptions);
447 					searcher.printResultSet(rs, limit);
448 
449 					if (interactiveOptions.displayResults())
450 						searcher.displayResults("Results:  " + interactiveOptions.getQueryImage(), rs, limit);
451 				} catch (final CmdLineException cle) {
452 					System.out.println("Syntax error:" + cle.getMessage());
453 				} catch (final Exception e) {
454 					e.printStackTrace();
455 					System.out.println("Error: " + e.getMessage());
456 				}
457 			}
458 		}
