41  public class BasicSearcherXmlRpcServlet {
42  	public static BasicSearcher<BasicSearcherOptions > searcher;
43  	public static BasicSearcherOptions options;
44  
45  	public List<Map<String, Object>> search(String localImageFile, boolean useNeighbours) throws Exception {
46  		return search(localImageFile, options.getLimit());
47  	}

49  	public List<Map<String, Object>> search(String localImageFile, int limit) throws Exception {
50  		List<Map<String, Object>> results = new ArrayList<Map<String, Object>>();
51  		List<DocidScore> rs = searcher.search(new File(localImageFile), null, options);
52  
53  		if (limit<=0) limit = rs.size();
54  
55  		for (DocidScore docidscore : rs) {
56  			File file = searcher.getFile(searcher.index,docidscore.first);
57  
58  			if (docidscore.second <= 0) break; //filter 0 results
59  
60  			Map<String, Object> entry = new HashMap<String, Object>();
61  			entry.put("name", file.toString());
62  			entry.put("score", docidscore.second);
63  			results.add(entry);
64  		}
65  
66  		return results;
67  	}
68  
69  	public List<Map<String, Object>> search(byte[] imageBytes) throws Exception {
70  		return search(imageBytes, options.getLimit());
71  	}
72  
73  	public List<Map<String, Object>> search(byte[] imageBytes, int limit) throws Exception {
74  		System.out.format("Search: %d byte image, limit:%d\n", imageBytes.length, limit);
75  		File tmp = File.createTempFile("imageterrier-xmlrpc-image", ".jpg");
76  		FileOutputStream fos = new FileOutputStream(tmp);
77  		fos.write(imageBytes);
78  		fos.close();
79  		List<Map<String, Object>> res = search(tmp.toString(), limit);
80  		System.out.println(tmp);
81  		//tmp.delete();
82  		System.out.println("Result: " + res);
83  		return res;
84  	}
