165 	protected <T extends QuantisedLocalFeature<?>> void configureRequest(SearchRequest request, QLFDocumentQuery<T> query) {
166 		TIntIntHashMap counts = new TIntIntHashMap();
167 		int max = 0;
168 		for (T ln : query.getDocument().getEntries()) {
169 			counts.adjustOrPutValue(ln.id, 1, 1);
170 			if (counts.get(ln.id) > max) max = counts.get(ln.id);			
171 		}
172 
173 		request.addMatchingModel("Matching", org.imageterrier.models.L1WeightingModel.class.getName());
174 		request.setControl("c_set", "true");
175 		request.setControl("c", "" + ((double)max / (double)query.getNumberOfTerms()));
176 	}
177 	
178 	protected <T extends QuantisedLocalFeature<?>> ResultSet search(Index index, QLFDocumentQuery<T> query) {
179 		Manager manager = new Manager(index);
180 		SearchRequest request = manager.newSearchRequest("foo");
181 		request.setQuery(query);
182 
183 		configureRequest(request, query);
184 
185 		manager.runPreProcessing(request);
186 		manager.runMatching(request);
187 		manager.runPostProcessing(request);
188 		manager.runPostFilters(request);
189 
190 		return request.getResultSet();
191 	}
