42  /**
43   * Options for indexing
44   * 
45   * @author Jonathon Hare <jsh2@ecs.soton.ac.uk>
46   */
47  public class BasicIndexerOptions {
48  	@Option(name = "--mode", aliases = "-m", usage = "input mode", required = true, handler=ProxyOptionHandler.class)
49  	private InputMode inputMode;
50  	private InputModeOptions inputModeOp;
51  	
52  	@Option(name = "--output", aliases = "-o", usage = "path at which to write index", required = true, metaVar = "path")
53  	private String filename;
54  
55  	@Option(name="--type", aliases="-t", required=false, usage="Choose index type", handler=ProxyOptionHandler.class)
56  	protected IndexType indexType = IndexType.BASIC;
57  	private IndexTypeOptions indexTypeOp = IndexType.BASIC.getOptions();
58  	
59  	@Option(name = "--verbose", aliases = "-v", usage = "print verbose output")
60  	boolean verbose = false;
61  	
62  	@Argument(required = true)
63  	private List<String> searchPaths = new ArrayList<String>();
64  	
65  	public boolean isVerbose() {
66  		return verbose;
67  	}
68  	
69  	public String getFilename() {
70  		return filename;
71  	}
72  	
73  	public IndexTypeOptions getIndexType() {
74  		return indexTypeOp;
75  	}
76  	
77  	public List<String> getSearchPaths() {
78  		return searchPaths;
79  	}
80  	
81  	public InputMode getInputMode() {
82  		return inputMode;
83  	}
84  	
85  	public InputModeOptions getInputModeOptions() {
86  		return inputModeOp;
87  	}
