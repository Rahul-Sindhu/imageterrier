29  package org.imageterrier.dsms;

31  import org.imageterrier.basictools.ApplicationSetupUtils;
32  import org.openimaj.math.geometry.point.Point2d;
33  import org.openimaj.math.geometry.transforms.AffineTransformModel;
34  import org.openimaj.math.model.EstimatableModel;
35  import org.terrier.matching.dsms.DocumentScoreModifier;
36  
37  /**
38   * A score modifier that can be applied to a position index with x and y
39   * coordinates, and works by fitting a {@link AffineTransformModel} to the
40   * matching visual term pairs.
41   * 
42   * If a transform that fits the point pairs is found the document score is set
43   * to the number of inliers, otherwise the score is zero.
44   * 
45   * @author Jonathon Hare <jsh2@ecs.soton.ac.uk>
46   */
47  public class AffineScoreModifier extends AbstractRANSACGeomModifier implements DocumentScoreModifier {
48  	/**
49  	 * The distance in pixels that points are allowed to move from their
50  	 * predicted position and still be considered a match.
51  	 */
52  	public static final String MODEL_TOLERANCE = "AffineScoreModifier.model_tolerance";
53  
54  	@Override
55  	public String getName() {
56  		return "AffineScoreModifier";
57  	}
58  
59  	@Override
60  	public AffineScoreModifier clone() {
61  		// new one, as we don't have state
62  		return new AffineScoreModifier();
63  	}
64  
65  	@Override
66  	public EstimatableModel<Point2d, Point2d> makeModel() {
67  		return new AffineTransformModel();
68  	}
69  
70  	@Override
71  	protected double getTolerance() {
72  		return ApplicationSetupUtils.getProperty(MODEL_TOLERANCE, 10.0f);
73  	}
74  }
