29  package org.imageterrier.features;
30  
31  import java.io.File;
32  import java.io.FileInputStream;
33  import java.io.IOException;
34  import java.util.concurrent.Callable;
35  
36  import org.apache.log4j.Logger;
37  import org.openimaj.feature.local.list.LocalFeatureList;
38  import org.openimaj.io.IOUtils;
39  import org.openimaj.tools.localfeature.options.LocalFeatureMode.LocalFeatureModeOp;
40  
41  public class FeatureTask implements Callable<File> {
42  	protected static Logger logger = Logger.getLogger(FeatureTask.class);
43  
44  	protected boolean forceRegeneration;
45  	protected File imageFile;
46  	protected LocalFeatureModeOp mode;
47  
48  	public FeatureTask(File imageFile, boolean forceRegeneration, LocalFeatureModeOp mode) {
49  		this.imageFile = imageFile;
50  		this.forceRegeneration = forceRegeneration;
51  		this.mode = mode;
52  	}
53  
54  	public File getFeatureFile() {
55  		return makeFeatureFilename(".fv");
56  	}
57  
58  	@Override
59  	public File call() throws Exception {
60  		return call(getFeatureFile());
61  	}
62  
63  	/**
64  	 * Process image and make loc file. Processed loc file will be created at
65  	 * saveLocation, unless one already existed alongside the image, in which
66  	 * case that one is returned.
67  	 * 
68  	 * @param filename
69  	 * @return filename
70  	 * @throws Exception
71  	 */
72  	public File call(File filename) throws Exception {
73  		if (filename.exists() && !forceRegeneration) {
74  			logger.info("Loading visual terms for " + imageFile.getName());
75  			return filename;
76  		}
77  
78  		// Get the keypoints
79  		try {
80  			logger.info("Generating keypoints for " + imageFile.getName());
81  			final LocalFeatureList<?> keys = mode.extract(loadImageData(imageFile));
82  			IOUtils.writeBinary(filename, keys);
83  		} catch (final Exception e) {
84  			System.err.println("Error processing image " + imageFile);
85  			return null;
86  		}
87  
88  		return filename;
89  	}
90  
91  	public static LocalFeatureList<?> computeFeatures(File imageFile, LocalFeatureModeOp mode) throws IOException {
92  		return mode.extract(loadImageData(imageFile));
93  	}
94  
95  	protected static byte[] loadImageData(File imageFile) throws IOException {
96  		if (imageFile.isDirectory())
97  			throw new RuntimeException("Unsupported operation, file " + imageFile.getAbsolutePath() + " is a directory");
98  		if (imageFile.length() > Integer.MAX_VALUE)
99  			throw new RuntimeException("Unsupported operation, file " + imageFile.getAbsolutePath() + " is too big");
100 
101 		FileInputStream in = null;
102 		final byte buffer[] = new byte[(int) imageFile.length()];
103 		try {
104 			in = new FileInputStream(imageFile);
105 			in.read(buffer);
106 		} catch (final Exception e) {
107 			throw new RuntimeException("Exception occured on reading file " + imageFile.getAbsolutePath(), e);
108 		} finally {
109 			if (in != null) {
110 				try {
111 					in.close();
112 				} catch (final Exception e) {
113 					throw new RuntimeException("Exception occured on closing file" + imageFile.getAbsolutePath(), e);
114 				}
115 			}
116 		}
117 		return buffer;
118 	}
119 
120 	protected File makeFeatureFilename(String extension) {
121 		String s = imageFile.toString();
122 
123 		s = s + extension;
124 
125 		return new File(s);
126 	}
