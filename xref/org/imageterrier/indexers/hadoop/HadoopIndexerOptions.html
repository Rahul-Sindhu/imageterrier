128 	/**
129 	 * Get the list of paths to build the index from. The paths should be
130 	 * sequence files containing quantised local features or images (depending
131 	 * on the mode). If a directory is given, it is automatically searched for
132 	 * internal sequence files starting with the name "part". This allows easy
133 	 * coupling to features generated from previous reducer passes (i.e. from
134 	 * the cluster-quantiser).
135 	 * 
136 	 * @return the input paths
137 	 * @throws IOException
138 	 */
139 	public Path[] getInputPaths() throws IOException {
140 		final List<Path> allPaths = new ArrayList<Path>();
141 
142 		for (final String inputPath : inputPaths) {
143 			final Path[] paths = SequenceFileUtility.getFilePaths(inputPath, "part");
144 
145 			for (final Path path : paths)
146 				allPaths.add(path);
147 		}
148 
149 		return allPaths.toArray(new Path[allPaths.size()]);
150 	}
