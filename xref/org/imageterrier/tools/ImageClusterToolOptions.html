241 	public Iterable<QLFDocument<QuantisedKeypoint>> getQueryIterator() {
242 		return new Iterable<QLFDocument<QuantisedKeypoint>>(){
243 			@Override
244 			public Iterator<QLFDocument<QuantisedKeypoint>> iterator() {
245 				final String[] fileList = ImageClusterToolOptions.this.queryDirectory.list(new FilenameFilter() {
246 					@Override
247 					public boolean accept(File file, String name) {
248 //						System.out.println(name+": " + quantisedFileRegex + ": "+ name.matches(quantisedFileRegex));
249 						if(ImageClusterToolOptions.this.quantisedFileRegex == null)
250 							return name.endsWith(ImageClusterToolOptions.this.quantisedExtention);
251 						else
252 							return name.matches(ImageClusterToolOptions.this.quantisedFileRegex);
253 					}
254 				});
255 				return new Iterator<QLFDocument<QuantisedKeypoint>>(){
256 					int index = 0;
257 					@Override
258 					public boolean hasNext() {
259 						return index < fileList.length;
260 					}
261 
262 					@Override
263 					public QLFDocument<QuantisedKeypoint> next() {
264 						String filename = fileList[index++];
265 						try {
266 							return ImageClusterToolOptions.this.getQuery(filename);
267 						} catch (IOException e) {
268 							if(this.hasNext()) return this.next();
269 							return null;
270 						}
271 					}
272 
273 					@Override
274 					public void remove() {
275 						throw new UnsupportedOperationException();
276 					}
277 				};
278 			}
279 			
280 		};

302 	public SimpleGraph<String, DefaultEdge> getImageGraph() {
303 		if(existingGraph != null)
304 			return existingGraph ;
305 		Index index = this.loadIndex();
306 		
307 		Manager manager = new Manager(index);
308 		Iterable<QLFDocument<QuantisedKeypoint>> queryIterator = this.getQueryIterator();
309 		SimpleGraph<String,DefaultEdge> imageGraph = new SimpleGraph<String,DefaultEdge>(DefaultEdge.class);
310 		
311 		for (QLFDocument<QuantisedKeypoint> query : queryIterator) {
312 			QLFDocumentQuery<QuantisedKeypoint> q = new QLFDocumentQuery<QuantisedKeypoint>(query);
313 			File docno = new File(q.getDocument().getProperty("docno"));
314 			String queryImageName = this.getImageFromQuery(docno.getName());
315 			System.out.println("Doing query: " + queryImageName);
316 			SearchRequest request = manager.newSearchRequest("foo");
317 			request.setQuery(q);
318 			this.getMatchingModelType().configureRequest(request, q);
319 			ApplicationSetup.setProperty("matching.dsms", this.getScoreModifierType().getScoreModifierClass(index.getInvertedIndex()));
320 			
321 			ApplicationSetup.setProperty("ignore.low.idf.terms","false");
322 			
323 			if (this.getLimit() == 0)
324 				ApplicationSetup.setProperty("matching.retrieved_set_size", ""+index.getCollectionStatistics().getNumberOfDocuments());
325 			else
326 				ApplicationSetup.setProperty("matching.retrieved_set_size", ""+this.getLimit());
327 			
328 			manager.runPreProcessing(request);
329 			manager.runMatching(request);
330 			manager.runPostProcessing(request);
331 			manager.runPostFilters(request);
332 			
333 			ResultSet resultsSet = request.getResultSet();
334 			int limit = resultsSet.getDocids().length;
335 			ScoredImageList results = new ScoredImageList(queryImageName);
336 			imageGraph.addVertex(queryImageName);
337 			for (int i=0; i<limit; i++) {
338 				int docid = resultsSet.getDocids()[i];
339 				String name;
340 				try {
341 					name = index.getMetaIndex().getItem("path", docid);
342 				} catch (IOException e) {
343 					continue;
344 				}
345 				if(resultsSet.getScores()[i] <= 0) continue;
346 				String resultImageName = this.getImageFromQuery(name);
347 				results.add(new ScoredImage(resultImageName,resultsSet.getScores()[i]));
348 				if(!resultImageName.equals(queryImageName)){
349 					imageGraph.addVertex(resultImageName);
350 					imageGraph.addEdge(queryImageName, resultImageName);
351 				}
352 			}
353 			try {
354 				this.appendResults(results);
355 			} catch (IOException e) {
356 				System.err.println("Results Writing Failed!");
357 			}
358 		}
359 		return imageGraph;
