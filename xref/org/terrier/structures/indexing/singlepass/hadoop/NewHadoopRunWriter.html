29  package org.terrier.structures.indexing.singlepass.hadoop;

32  
33  import org.apache.hadoop.io.Text;
34  import org.apache.hadoop.mapreduce.Mapper;
35  import org.terrier.compression.MemorySBOS;
36  import org.terrier.structures.indexing.singlepass.Posting;
37  import org.terrier.structures.indexing.singlepass.RunWriter;
38  import org.terrier.utility.io.WrappedIOException;

40  /** 
41   * RunWriter for the MapReduce indexer
42   * Updated for the new API
43   *  
44   * @author Richard McCreadie and Craig Macdonald
45   * @author Jonathon Hare
46   *  
47   * @version $Revision: 1.0 $ 
48   * @param <VALUEIN>  
49   */
50  public class NewHadoopRunWriter<VALUEIN> extends RunWriter {
51  	/** output collector of Map task */
52  	protected Mapper<Text,VALUEIN,NewSplitEmittedTerm,MapEmittedPostingList>.Context context = null;
53  	
54  	/** map task id that is being flushed */
55  	protected String mapId;
56  	
57  	/** flushNo is the number of times this map task is being flushed */
58  	protected int flushNo;
59  	
60  	/** current split id */
61  	protected int splitId;
62  	
63  	/** Create a new HadoopRunWriter, specifying the output collector of the map task
64  	 * the run number and the flush number.
65  	 * @param context where to emit the posting lists to
66  	 * @param mapId the task id of the map currently being processed
67  	 * @param splitId the id of the split
68  	 * @param flushNo the number of times that this map task has flushed
69  	 */
70  	public NewHadoopRunWriter(Mapper<Text,VALUEIN,NewSplitEmittedTerm,MapEmittedPostingList>.Context context, String mapId, int splitId, int flushNo)
71  	{
72  		this.context = context;
73  		this.mapId = mapId;
74  		this.flushNo = flushNo;
75  		this.splitId = splitId;
76  		this.info = "NewHadoopRunWriter(Map "+ mapId +", flush "+flushNo+")"; 
77  	}
78  	
79  	@Override
80  	public void beginWrite(int maxSize, int size) throws IOException
81  	{}
82  	
83  	/** Write the posting to the output collector
84  	 */
85  	@Override
86  	public void writeTerm(final String term, final Posting post) throws IOException
87  	{	
88  		final MemorySBOS Docs = post.getDocs();
89  		Docs.pad();
90  		//get the posting array buffer
91  		byte[] buffer = new byte[Docs.getMOS().getPos()+1];
92  		System.arraycopy(Docs.getMOS().getBuffer(), 0, 
93  				buffer, 0, 
94  				Math.min(Docs.getMOS().getBuffer().length, Docs.getMOS().getPos()+1));
95  		
96  		//emit the term and its posting list
97  		try {
98  			context.write(
99  					NewSplitEmittedTerm.createNewTerm(term, splitId, flushNo), 
100 					MapEmittedPostingList.create_Hadoop_WritableRunPostingData(
101 							mapId,
102 							flushNo, 
103 							splitId,
104 							buffer,
105 							post.getDocF(), post.getTF()));
106 		} catch (InterruptedException e) {
107 			throw new WrappedIOException(e);
108 		}
109 	}
110 	
111 	@Override
112 	public void finishWrite() throws IOException
113 	{}
114 	
115 	/** This RunWriter does not require that the output be sorted.
116 	  */
117 	@Override
118 	public boolean writeSorted()
119 	{
120 		return false;
121 	}
122 }
