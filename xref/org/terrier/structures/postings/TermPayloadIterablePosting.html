29  package org.terrier.structures.postings;

31  import java.io.DataInput;
32  import java.io.DataOutput;
33  import java.io.IOException;
34  
35  import org.terrier.compression.BitIn;
36  import org.terrier.structures.DocumentIndex;
37  import org.terrier.structures.TermPayloadCoordinator;
38  import org.terrier.structures.postings.BasicIterablePosting;
39  import org.terrier.structures.postings.WritablePosting;

41  /**
42   * An extension to {@link BasicIterablePosting} that allows payload
43   * data to be stored with each posting. 
44   * 
45   * @author Jonathon Hare <jsh2@ecs.soton.ac.uk>
46   *
47   * @param <PAYLOAD> type of payload
48   */
49  public class TermPayloadIterablePosting<PAYLOAD> extends BasicIterablePosting implements TermPayloadPosting<PAYLOAD> {
50  	/**
51  	 * This stores info about the payload
52  	 */
53  	protected TermPayloadCoordinator<PAYLOAD> payloadConf;
54  	
55  	protected PAYLOAD[] payloads;
56  	
57  	/**
58  	 * Create a new posting iterator
59  	 */
60  	public TermPayloadIterablePosting() { super(); }
61  	
62  	/** Create a new posting iterator
63  	 * @param _bitFileReader BitIn to read the postings from
64  	 * @param _numEntries number of postings in the list
65  	 * @param doi document index to use to satisfy getDocumentLength()
66  	 * @throws IOException thrown in an IO exception occurs
67  	 */
68  	public TermPayloadIterablePosting(BitIn _bitFileReader, int _numEntries, DocumentIndex doi) throws IOException {
69  		super(_bitFileReader, _numEntries, doi);
70  	}
71  
72  	/**
73  	 * Set the payload coordinator
74  	 * @param payloadConf the coordinator
75  	 */
76  	public void setPayloadConfig(TermPayloadCoordinator<PAYLOAD> payloadConf) {
77  		this.payloadConf = payloadConf;
78  	}
79  	
80  	@Override
81  	public int next() throws IOException {
82  		if (numEntries-- == 0) return EOL;
83  		
84  		id = bitFileReader.readGamma() + id; //id is the docid
85  		tf = bitFileReader.readUnary(); //tf is how many times the term occurs in the doc
86  		
87  		//TODO: this has a memory allocation for every posting in the posting list. can we reuse an array?
88  		payloads = payloadConf.makePayloadArray(tf);
89  		
90  		for (int k=0; k<tf; k++) {
91  			payloads[k] = payloadConf.readPayload(bitFileReader); 
92  		}
93  		
94  		return id;
95  	}
96  	
97  	@Override
98  	public PAYLOAD[] getPayloads() {
99  		return payloads;
100 	}
101 
102 	@Override
103 	public void readFields(DataInput in) throws IOException {
104 		super.readFields(in);
105 		
106 		payloads = payloadConf.makePayloadArray(tf); 
107 		
108 		for(int i=0;i<tf;i++) {
109 			payloads[i] = payloadConf.readPayload(in);
110 		}
111 	}
112 
113 	@Override
114 	public void write(DataOutput out) throws IOException {
115 		super.write(out);
116 		
117 		for(PAYLOAD payload : payloads) {
118 			payloadConf.writePayload(out, payload);
119 		}
120 	}
121 
122 	@Override
123 	public WritablePosting asWritablePosting() {
124 		PAYLOAD[] newPayloads = payloads.clone();
125 		
126 		for(int row=0; row<payloads.length; row++)
127 			newPayloads[row] = payloadConf.clone(payloads[row]);
128 		
129 		return new TermPayloadPostingImpl<PAYLOAD>(getId(), getFrequency(), newPayloads, payloadConf);
130 	}
131 
132 	@Override
133 	public String toString()
134 	{
135 		String pos = "";
136 		for (PAYLOAD payload : payloads) pos += "(" + payload + "),";
137 		pos.subSequence(0, pos.length() - 1);
138 		
139 		return "(" + id + "," + tf + ",B[" + pos + "])";
140 	}
